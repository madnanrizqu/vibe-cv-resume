% Custom commands

\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-1pt}}
  }
}

% MODIFIED: row 2 uses regular text instead of \textit{#3}
% Allows first line to wrap if too long, keeps dates aligned properly
\newcommand{\resumeSubheading}[4]{
  \vspace{4pt}\item
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & {\color{dark-grey}\small #2}\\
    \end{tabular*}\vspace{1pt}
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}
      #3 & {\color{dark-grey} \small #4}\\
    \end{tabular*}\vspace{-4pt}
}

\newcommand{\resumeSubSubheading}[2]{
    \item
    \begin{tabular*}{\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeProjectHeading}[1]{

    \small{#1}
    \vspace{-4pt}
}

\newcommand{\resumeSubItem}[1]{\resumeItem{#1}\vspace{-4pt}}

\newcommand{\resumeSimpleEntry}[2]{

    \end{tabular*}\vspace{-16pt}
}

}
